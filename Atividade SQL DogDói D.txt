WHERE v.ESPECIALIDADE = 'Dermatologista'

SELECT * FROM PACIENTE p 
WHERE p.IDADE BETWEEN 2 AND 5

SELECT * FROM AGENDAMENTO a 
WHERE a.DATA_HORA BETWEEN '2024-01-01 00:00:00.000' AND '2024-03-31 23:59:59.000'

SELECT * FROM MEDICAMENTOS m 
WHERE m.PRECO > 25

SELECT * FROM RECEITA r 
WHERE r.ID_AGENDAMENTO = (SELECT v.ID_VETERINARIO FROM VETERINARIO v WHERE v.NOME = 'Dr. Ana Souza')
AND DATA_EMISSAO > '2024-06-01'

SELECT * FROM PACIENTE p 
WHERE p.ID_TUTOR = (SELECT t.ID_TUTOR FROM TUTOR t WHERE t.NOME LIKE 'Maria Silva')

SELECT * FROM ITENSRECEITA i 
WHERE i.ID_MEDICAMENTO = (SELECT m.ID_MEDICAMENTO FROM MEDICAMENTOS m WHERE m.NOME LIKE 'Anti-inflamatório Y')

SELECT 
    TUTOR.id_tutor,
    TUTOR.nome,
    (SELECT COUNT(*) 
     FROM PACIENTE 
     WHERE PACIENTE.id_tutor = TUTOR.id_tutor) AS total_pacientes
FROM 
    TUTOR

SELECT A.ID_AGENDAMENTO, A.DATA_HORA , A.MOTIVO,
       A.ID_PACIENTE , A.ID_VETERINARIO,
      (SELECT p.PESO FROM PACIENTE p WHERE P.ID_PACIENTE = A.ID_PACIENTE),
      (SELECT p.NOME FROM PACIENTE p WHERE P.ID_PACIENTE = A.ID_PACIENTE)
FROM AGENDAMENTO a 
WHERE a.DATA_HORA BETWEEN '2024-04-01 10:00:00.000' AND '2024-06-30 10:00:00.000'
